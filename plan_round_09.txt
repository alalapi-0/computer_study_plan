from ai_prep_tool import filter_records, dedup_records, build_summary

def test_filter_records_keeps_keyword():
    records = ["ok sample", "bad sample", "ok again"]

    assert len(result) == 2

def test_dedup_records_removes_duplicates():
    records = ["a", "a", "b"]
    result = dedup_records(records, "txt")
    assert result == ["a", "b"]

def test_build_summary_counts():
    summary = build_summary("in.txt", "out.txt", 5, 3, 2)
    assert summary["raw_count"] == 5
    assert summary["filtered_count"] == 3
    assert summary["final_count"] == 2
