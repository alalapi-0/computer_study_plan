import csv
import torch
from torch.utils.data import Dataset, DataLoader

def simple_tokenize(text: str):
    return text.lower().split()

special_tokens = ["<PAD>", "<UNK>"]
vocab = {tok: i for i, tok in enumerate(special_tokens)}

rows = []
with open("../week1/texts.csv", "r", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for row in reader:
        rows.append(row)
        for token in simple_tokenize(row["text"]):
            if token not in vocab:
                vocab[token] = len(vocab)

def encode(text: str, max_len=6):
    ids = [vocab.get(tok, vocab["<UNK>"]) for tok in simple_tokenize(text)]
    ids = ids[:max_len]
    ids += [vocab["<PAD>"]] * (max_len - len(ids))
    return ids

class TextDataset(Dataset):
    def __init__(self, rows):
        self.rows = rows

    def __len__(self):
        return len(self.rows)

    def __getitem__(self, idx):
        row = self.rows[idx]
        x = torch.tensor(encode(row["text"]), dtype=torch.long)
        y = torch.tensor(int(row["label"]), dtype=torch.long)
        return x, y

dataset = TextDataset(rows)
loader = DataLoader(dataset, batch_size=2, shuffle=True)

for X, y in loader:
    print("X shape:", X.shape)
    print("y shape:", y.shape)
    print(X)
    print(y)

练习 11：写最小文本分类训练脚本

新建：

touch train_text_classifier.py

写一个最小版：

import csv
import torch
from torch import nn
from torch.utils.data import Dataset, DataLoader

def simple_tokenize(text: str):
    return text.lower().split()

special_tokens = ["<PAD>", "<UNK>"]
vocab = {tok: i for i, tok in enumerate(special_tokens)}

rows = []
with open("sentiment.csv", "r", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for row in reader:
        rows.append(row)
        for token in simple_tokenize(row["text"]):
            if token not in vocab:
                vocab[token] = len(vocab)

max_len = 6

def encode(text: str):
    ids = [vocab.get(tok, vocab["<UNK>"]) for tok in simple_tokenize(text)]
    ids = ids[:max_len]
    ids += [vocab["<PAD>"]] * (max_len - len(ids))
    return ids

class TextDataset(Dataset):
    def __init__(self, rows):
        self.rows = rows

    def __len__(self):
        return len(self.rows)

    def __getitem__(self, idx):
        row = self.rows[idx]
        x = torch.tensor(encode(row["text"]), dtype=torch.long)
        y = torch.tensor(int(row["label"]), dtype=torch.long)
        return x, y

dataset = TextDataset(rows)
loader = DataLoader(dataset, batch_size=2, shuffle=True)

class TextClassifier(nn.Module):
    def __init__(self, vocab_size, embed_dim, num_classes):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, embed_dim, padding_idx=0)
        self.fc = nn.Linear(embed_dim, num_classes)

    def forward(self, x):
        emb = self.embedding(x)           # [B, T, D]
        pooled = emb.mean(dim=1)          # [B, D]
        logits = self.fc(pooled)          # [B, C]
        return logits

model = TextClassifier(len(vocab), 8, 2)
loss_fn = nn.CrossEntropyLoss()
optimizer = torch.optim.SGD(model.parameters(), lr=0.1)

for epoch in range(10):
    model.train()
    total_loss = 0.0
    for X, y in loader:
        pred = model(X)
        loss = loss_fn(pred, y)

        loss.backward()
        optimizer.step()
        optimizer.zero_grad()

        total_loss += loss.item()

    print(f"epoch={epoch+1}, loss={total_loss:.4f}")

这里最重要的不是模型“高级”，而是你第一次真正拿文本任务跑通了 PyTorch 训练循环。训练循环里的 loss.backward()、optimizer.step()、optimizer.zero_grad()，依然是 PyTorch 官方上一轮已经强调过的核心三步。(pytorch.org￼  ￼)
